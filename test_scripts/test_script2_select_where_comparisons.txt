CREATE DATABASE comparisons_demo;
USE comparisons_demo;
CREATE TABLE users (id int INDEXED, name string, age int);
INSERT INTO users (id, name, age) VALUE (1, "Ann", 17), (2, "Bob", 20), (3, "Alice", 30), (4, "Carl", 25);
SELECT * FROM users;
SELECT (id, name) FROM users WHERE age >= 20;
SELECT (name, age) FROM users WHERE age < 25;
SELECT (id, name) FROM users WHERE name < "Carl";
